2.You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps) Input: abcadeecfb Output: a=2 b=2 c=2 d=1 e=2 f=1" 

            let  c = "abcadeecfb";
            let e = c.split("")
            let num = new Map();

            for (let i=0 ; i< e.length; i++)
            {
                if(num.has(e[i]) == true){
                    let value = num.get(e[i])
                    num.set(e[i],value+1);
                }
                else{
                    num.set(e[i],1)
                }
            }
            for(let i of num){
                console.log(i[0] + "=" + i[1]);
            }
